        function searchAddrFromCoords(coords, callback) {
            geocoder.coord2RegionCode(coords.getLng(), coords.getLat(), callback);  // 좌표로 행정동 주소 정보를 요청합니다   
        }

        function searchDetailAddrFromCoords(coords, callback) {   
            geocoder.coord2Address(coords.getLng(), coords.getLat(), callback);// 좌표로 법정동 상세 주소 정보를 요청합니다
        }

        function displayCenterInfo(result, status){
